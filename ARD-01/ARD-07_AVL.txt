const int photocellPin = A0;  // Definimos entrada/salida Anal칩gica para lecturas del Fotoresistor
int sensorValue = 0;          // Inicializamos el valor del sensor
const int led = 7;
void setup() {
  pinMode(led, OUTPUT);
  Serial.begin(9600);  //Inicializamos comunicaci칩n serial con el Monitor Serial de Arduino IDE
  //N칩tese que el Pin An치logo no necesita especificarse como Entrada/Salida
}
void loop() {
  sensorValue = analogRead(photocellPin);  //se toman las lecturas del sensor
  Serial.print("Lectura = ");              //Se muestra en el Monitor Serial
  Serial.println(sensorValue);
  delay(1000);
  if (sensorValue < 100 || sensorValue > 900) {
    digitalWrite(led, HIGH);
  } else {
    digitalWrite(led, LOW);
  }
}
